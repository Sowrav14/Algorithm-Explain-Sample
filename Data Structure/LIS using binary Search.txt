void dfs(int vertex){
    vis[vertex] = 1;
    int cur, flag;
    flag = 0;
    int it = lower_bound(lis.begin(), lis.end(), v[vertex]) - lis.begin();
    if(it != lis.size()){
        cur = lis[it];
        flag = 1;
        lis[it] = v[vertex];

    } else {
        lis.pb(v[vertex]);
    }
    for(auto child : g[vertex]){
        if(!vis[child]){
            dfs(child);
        }
    }
    ans[vertex] = lis.size();
    if(flag){
        lis[it] = cur;
    } else{
        lis.pop_back();
    }
}
